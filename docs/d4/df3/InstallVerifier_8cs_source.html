74 private static bool HashMatchesFile(string path, byte[] hash)
75 {
76 using MD5 md5 = MD5.Create();
77 using FileStream stream = File.OpenRead(path);
78 return hash.SequenceEqual(md5.ComputeHash(stream));
79 }

81 private static byte[] ToByteArray(string hexString, bool forceLowerCase = true)
82 {
83 if (forceLowerCase)
84 {
85 hexString = hexString.ToLower();
86 }
87 byte[] retval = new byte[hexString.Length / 2];
88 for (int i = 0; i < hexString.Length; i += 2)
89 {
90 retval[i / 2] = Convert.ToByte(hexString.Substring(i, 2), 16);

109 private static DistributionPlatform DetectPlatform(out string detectionDetails)
110 {
111 if (Program.LaunchParameters.ContainsKey("-steam"))
112 {
113 detectionDetails = "-steam launch parameter";
114 return DistributionPlatform.Steam;
115 }
116 if (Directory.GetCurrentDirectory().Contains("steamapps", StringComparison.OrdinalIgnoreCase))
117 {
118 detectionDetails = "CWD is /steamapps/";
119 return DistributionPlatform.Steam;
120 }
121 if (!ObtainVanillaExePath(out var vanillaSteamAPIDir, out vanillaExePath))
122 {
123 detectionDetails = VanillaExe + " or " + CheckExe + " not found nearby";
124 return DistributionPlatform.Steam;
125 }
126 if (Platform.IsOSX)
127 {
128 vanillaSteamAPIDir = Path.Combine(Directory.GetParent(vanillaSteamAPIDir).FullName, "MacOS", "osx");
129 }
130 if (File.Exists(Path.Combine(vanillaSteamAPIDir, vanillaSteamAPI)))
131 {
132 detectionDetails = vanillaSteamAPI + " found";
133 return DistributionPlatform.Steam;
134 }
135 detectionDetails = Path.GetFileName(vanillaExePath) + " found, no steam files or directories nearby.";
136 return DistributionPlatform.GoG;
137 }

181 private static void CheckSteam()
182 {
183 if (!HashMatchesFile(steamAPIPath, steamAPIHash))
184 {
185 Utils.OpenToURL("https://terraria.org");
186 ErrorReporting.FatalExit(Language.GetTextValue("tModLoader.SteamAPIHashMismatch"));
187 }
188 else if (!Main.dedServ)
189 {
190 TerrariaSteamClient.LaunchResult result = TerrariaSteamClient.Launch();
191 switch (result)
192 {
193 case TerrariaSteamClient.LaunchResult.ErrClientProcDied:
194 ErrorReporting.FatalExit("The terraria steam client process exited unexpectedly");
195 break;
196 case TerrariaSteamClient.LaunchResult.ErrSteamInitFailed:
197 ErrorReporting.FatalExit(Language.GetTextValue("tModLoader.SteamInitFailed"));
198 break;
199 case TerrariaSteamClient.LaunchResult.ErrNotInstalled:
200 ErrorReporting.FatalExit(Language.GetTextValue("tModLoader.TerrariaNotInstalled"));
201 break;
202 case TerrariaSteamClient.LaunchResult.ErrInstallOutOfDate:
203 ErrorReporting.FatalExit(Language.GetTextValue("tModLoader.TerrariaOutOfDateMessage"));
204 break;
205 default:
206 throw new Exception("Unsupported result type: " + result);
207 case TerrariaSteamClient.LaunchResult.Ok:
208 break;
209 }
210 }
211 }

223 File.Copy(vanillaExePath, pathToCheckExe);
224 }
