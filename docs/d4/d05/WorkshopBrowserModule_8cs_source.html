167 public async IAsyncEnumerable<ModDownloadItem> QueryBrowser(QueryParameters queryParams, [EnumeratorCancellation] CancellationToken token = default(CancellationToken))
168 {
169 if (!SteamedWraps.SteamAvailable)
170 {
171 yield break;
172 }
173 if (queryParams.searchModIds != null && queryParams.searchModIds.Any())
174 {
175 List<string> missingMods;
176 foreach (ModDownloadItem item4 in DirectQueryItems(queryParams, out missingMods))
177 {
178 yield return item4;
179 }
180 yield break;
181 }
182 switch (queryParams.updateStatusFilter)
183 {
184 case UpdateFilter.All:
185 await foreach (ModDownloadItem item2 in WorkshopHelper.QueryHelper.QueryWorkshop(queryParams, token))
186 {
187 if (CachedInstalledModDownloadItems.Contains(item2) || intermediateInstallStateMods.Contains(item2.ModName))
188 {
189 item2.UpdateInstallState();
190 }
191 yield return item2;
192 }
193 break;
194 case UpdateFilter.Available:
195 await foreach (ModDownloadItem item3 in WorkshopHelper.QueryHelper.QueryWorkshop(queryParams, token))
196 {
197 if (!CachedInstalledModDownloadItems.Contains(item3) && !intermediateInstallStateMods.Contains(item3.ModName))
198 {
199 yield return item3;
200 }
201 }
202 break;
203 case UpdateFilter.UpdateOnly:
204 foreach (ModDownloadItem item5 in CachedInstalledModDownloadItems.Where((ModDownloadItem item) => item.NeedUpdate))
205 {
206 yield return item5;
207 }
208 break;
209 case UpdateFilter.InstalledOnly:
210 foreach (ModDownloadItem cachedInstalledModDownloadItem in CachedInstalledModDownloadItems)
211 {
212 yield return cachedInstalledModDownloadItem;
213 }
214 break;
215 }
