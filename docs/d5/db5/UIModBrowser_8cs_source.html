364 public override void Update(GameTime gameTime)
365 {
366 base.Update(gameTime);
367 lock (modSlugsToUpdateInstallInfo)
368 {
369 foreach (UIModDownloadItem item in ModList.ReceivedItems.Where((UIModDownloadItem d) => modSlugsToUpdateInstallInfo.Contains(d.ModDownload.ModName)))
370 {
371 item.ModDownload.UpdateInstallState();
372 item.UpdateInstallDisplayState();
373 }
374 modSlugsToUpdateInstallInfo.Clear();
375 }
376 if (DebounceTimer != null && DebounceTimer.Elapsed >= MinTimeBetweenUpdates)
377 {
378 DebounceTimer.Stop();
379 DebounceTimer = null;
380 }
381 if (UpdateNeeded && DebounceTimer == null)
382 {
383 UpdateNeeded = false;
384 PopulateModBrowser();
385 DebounceTimer = new Stopwatch();
386 DebounceTimer.Start();
387 }

402 internal async void DownloadModsAndReturnToBrowser(List<ModPubId_t> modIds)
403 {
404 List<string> missingMods;
405 List<ModDownloadItem> downloadsQueried = SocialBackend.DirectQueryItems(new QueryParameters
406 {
407 searchModIds = modIds.ToArray()
408 }, out missingMods);
409 if (!(await DownloadMods(downloadsQueried)))
410 {
411 return;
412 }
413 if (missingMods.Any())
414 {
415 Interface.errorMessage.Show(Language.GetTextValue("tModLoader.MBModsNotFoundOnline", string.Join(",", missingMods)), 10007);
416 return;
417 }
418 Main.QueueMainThreadAction(delegate
419 {
420 Main.menuMode = 10007;
421 });
422 }

439 internal static async Task<bool> DownloadMods(IEnumerable<ModDownloadItem> mods, int previousMenuId, Action setReloadRequred = null, Action<ModDownloadItem> onNewModInstalled = null)
440 {
441 HashSet<ModDownloadItem> set = mods.ToHashSet();
442 Interface.modBrowser.SocialBackend.GetDependenciesRecursive(set);
443 List<ModDownloadItem> fullList = ModDownloadItem.NeedsInstallOrUpdate(set).ToList();
444 if (!fullList.Any())
445 {
446 return true;
447 }
448 HashSet<string> downloadedList = new HashSet<string>();
449 try
450 {
451 UIWorkshopDownload ui = new UIWorkshopDownload();
452 Main.menuMode = 888;
453 Main.MenuUI.SetState(ui);
454 await Task.Yield();
455 foreach (ModDownloadItem mod in fullList)
456 {
457 bool isInstalled = mod.IsInstalled;
458 if (ModLoader.TryGetMod(mod.ModName, out var loadedMod))
459 {
460 loadedMod.Close();
461 mod.Installed = null;
462 setReloadRequred?.Invoke();
463 }
464 Interface.modBrowser.SocialBackend.DownloadItem(mod, ui);
465 if (!isInstalled)
466 {
467 onNewModInstalled?.Invoke(mod);
468 }
469 downloadedList.Add(mod.ModName);
470 }
471 return true;
472 }
473 catch (Exception e)
474 {
475 LogModBrowserException(e, previousMenuId);
476 return false;
477 }
478 finally
479 {
480 ModOrganizer.LocalModsChanged(downloadedList, isDeletion: false);
481 }
482 }
