170 internal static string InsertMenu(out Action onClick)

187 if (styleText == null || bossBarStyles.Count == 1)
188 {
189 styleText = Language.GetTextValue("tModLoader.BossBarStyleNoOptions");

191 onClick = delegate

194 };
195 return Language.GetTextValue("tModLoader.BossBarStyle", styleText);

198 public static bool PreDraw(SpriteBatch spriteBatch, BigProgressBarInfo info, ref BossBarDrawParams drawParams)
199 {
200 int index = info.npcIndexToAimAt;
201 if (index < 0 || index > Main.maxNPCs)
202 {
203 return false;
204 }
205 NPC npc = Main.npc[index];
206 ModBossBar bossBar;
207 bool isModded = NpcToBossBar(npc, out bossBar);
208 if (isModded)
209 {
210 drawParams.BarTexture = GetTexture(bossBar).Value;
211 }
212 bool modify = true;
213 foreach (GlobalBossBar globalBossBar in globalBossBars)
214 {
215 modify &= globalBossBar.PreDraw(spriteBatch, npc, ref drawParams);
216 }
217 if (modify && isModded)
218 {
219 modify = bossBar.PreDraw(spriteBatch, npc, ref drawParams);
220 }
221 return modify;
222 }

224 public static void PostDraw(SpriteBatch spriteBatch, BigProgressBarInfo info, BossBarDrawParams drawParams)
225 {
226 int index = info.npcIndexToAimAt;
227 if (index < 0 || index > Main.maxNPCs)
228 {
229 return;
230 }
231 NPC npc = Main.npc[index];
232 if (NpcToBossBar(npc, out var bossBar))
233 {
234 bossBar.PostDraw(spriteBatch, npc, drawParams);
235 }
236 foreach (GlobalBossBar globalBossBar in globalBossBars)
237 {
238 globalBossBar.PostDraw(spriteBatch, npc, drawParams);
239 }
