50 internal static void Load(string path, bool isCloudSave)
51 {
52 customDataFail = null;
53 path = Path.ChangeExtension(path, ".twld");
54 if (FileUtilities.Exists(path, isCloudSave))
55 {
56 byte[] buf = FileUtilities.ReadAllBytes(path, isCloudSave);
57 if (buf[0] != 31 || buf[1] != 139)
58 {
59 throw new IOException(Path.GetFileName(path) + ":: File Corrupted during Last Save Step. Aborting... ERROR: Missing NBT Header");
60 }
61 TagCompound tag = TagIO.FromStream(buf.ToMemoryStream());
62 TileIO.LoadBasics(tag.GetCompound("tiles"));
63 TileIO.LoadContainers(tag.GetCompound("containers"));
64 LoadNPCs(tag.GetList<TagCompound>("npcs"));
65 try
66 {
67 TileIO.LoadTileEntities(tag.GetList<TagCompound>("tileEntities"));
68 }
69 catch (CustomModDataException ex)
70 {
71 customDataFail = ex;
72 throw;
73 }
74 LoadChestInventory(tag.GetList<TagCompound>("chests"));
75 LoadNPCKillCounts(tag.GetList<TagCompound>("killCounts"));
76 LoadNPCBestiaryKills(tag.GetList<TagCompound>("bestiaryKills"));
77 LoadNPCBestiarySights(tag.GetList<TagCompound>("bestiarySights"));
78 LoadNPCBestiaryChats(tag.GetList<TagCompound>("bestiaryChats"));
79 LoadAnglerQuest(tag.GetCompound("anglerQuest"));
80 LoadTownManager(tag.GetList<TagCompound>("townManager"));
81 try
82 {
83 LoadModData(tag.GetList<TagCompound>("modData"));
84 }
85 catch (CustomModDataException ex2)
86 {
87 customDataFail = ex2;
88 throw;
89 }
90 LoadAlteredVanillaFields(tag.GetCompound("alteredVanillaFields"));

733 internal static void ReadWorldHeader(WorldFileData data)
734 {
735 string path = Path.ChangeExtension(data.Path, ".twld");
736 bool isCloudSave = data.IsCloudSave;
737 if (!FileUtilities.Exists(path, isCloudSave))
738 {
739 return;
740 }
741 try
742 {
743 using Stream stream = (isCloudSave ? ((Stream)new MemoryStream(SocialAPI.Cloud.Read(path))) : ((Stream)new FileStream(path, FileMode.Open)));
744 using BinaryReader reader = new BigEndianReader(new GZipStream(stream, CompressionMode.Decompress));
745 if (reader.ReadByte() != 10)
746 {
747 throw new IOException("Root tag not a TagCompound");
748 }
749 TagIO.ReadTagImpl(8, reader);
750 if (reader.ReadByte() == 10 && !((string)TagIO.ReadTagImpl(8, reader) != "0header"))
751 {
752 LoadWorldHeader(data, (TagCompound)TagIO.ReadTagImpl(10, reader));
753 }
754 }
755 catch (Exception ex)
756 {
757 Logging.tML.Warn((object)$"Error reading .twld header from: {path} (IsCloudSave={isCloudSave})", ex);
758 }
