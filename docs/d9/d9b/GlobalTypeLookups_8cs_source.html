168 public static TGlobal[] CachedFilter(TGlobal[] arr, Predicate<TGlobal> filter)
169 {
170 TGlobal[] buf = ArrayPool<TGlobal>.Shared.Rent(arr.Length);
171 TGlobal[] array;
172 try
173 {
174 int i = 0;
175 array = arr;
176 foreach (TGlobal g in array)
177 {
178 if (filter(g))
179 {
180 buf[i++] = g;
181 }
182 }
183 if (i == arr.Length)
184 {
185 array = arr;
186 }
187 else
188 {
189 lock (_cache)
190 {
191 Memory<TGlobal> filtered = buf.AsMemory().Slice(0, i);
192 if (_cache.TryGetValue(filtered, out var cached))
193 {
194 array = cached;
195 }
196 else
197 {
198 TGlobal[] result = filtered.ToArray();
199 _cache.Add(result, result);
200 array = result;
201 }
202 }
203 }
204 }
205 finally
206 {
207 ArrayPool<TGlobal>.Shared.Return(buf, clearArray: true);
208 }
209 return array;
210 }
