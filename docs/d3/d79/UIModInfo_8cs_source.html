458 public override void Update(GameTime gameTime)
459 {
460 if (!_loading && _ready)
461 {
462 _modInfo.SetText(_info);
463 if (!string.IsNullOrEmpty(_url))
464 {
465 _uIElement.Append(_modHomepageButton);
466 }
467 if (!string.IsNullOrEmpty(_publishedFileId.m_ModPubId))
468 {
469 _uIElement.Append(_modSteamButton);
470 }
471 if (_localMod != null)
472 {
473 bool realDeleteButton = ModLoader.Mods.All((Mod x) => x.Name != _localMod.Name);
474 _uIElement.AddOrRemoveChild(_deleteButton, realDeleteButton);
475 _uIElement.AddOrRemoveChild(_fakeDeleteButton, !realDeleteButton);
476 _uIElement.AddOrRemoveChild(extractLocalizationButton, !realDeleteButton);
477 _uIElement.AddOrRemoveChild(fakeExtractLocalizationButton, realDeleteButton);
478 extractLocalizationButton.SetText(Language.GetTextValue("tModLoader.ModInfoExtractLocalization"));
479 _uIElement.Append(_extractButton);
480 }
481 Recalculate();
482 _modInfo.RemoveChild(_loaderElement);
483 _ready = false;
484 }
485 base.Update(gameTime);
