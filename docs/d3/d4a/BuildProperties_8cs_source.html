141 internal static BuildProperties ReadBuildFile(string modDir)
142 {
143 string propertiesFile = modDir + Path.DirectorySeparatorChar + "build.txt";
144 string descriptionfile = modDir + Path.DirectorySeparatorChar + "description.txt";
145 BuildProperties properties = new BuildProperties();
146 if (!File.Exists(propertiesFile))
147 {
148 return properties;
149 }
150 if (File.Exists(descriptionfile))
151 {
152 properties.description = File.ReadAllText(descriptionfile);
153 }
154 string[] array = File.ReadAllLines(propertiesFile);
155 foreach (string line in array)
156 {
157 if (string.IsNullOrWhiteSpace(line))
158 {
159 continue;
160 }
161 int split = line.IndexOf('=');
162 string property = line.Substring(0, split).Trim();
163 string value = line.Substring(split + 1).Trim();
164 if (value.Length == 0)
165 {
166 continue;
167 }
168 switch (property)
169 {
170 case "dllReferences":
171 properties.dllReferences = ReadList(value).ToArray();
172 break;
173 case "modReferences":
174 properties.modReferences = ReadList(value).Select(ModReference.Parse).ToArray();
175 break;
176 case "weakReferences":
177 properties.weakReferences = ReadList(value).Select(ModReference.Parse).ToArray();
178 break;
179 case "sortBefore":
180 properties.sortBefore = ReadList(value).ToArray();
181 break;
182 case "sortAfter":
183 properties.sortAfter = ReadList(value).ToArray();
184 break;
185 case "author":
186 properties.author = value;
187 break;
188 case "version":
189 properties.version = new Version(value);
190 break;
191 case "displayName":
192 properties.displayName = value;
193 break;
194 case "homepage":
195 properties.homepage = value;
196 break;
197 case "noCompile":
198 properties.noCompile = string.Equals(value, "true", StringComparison.OrdinalIgnoreCase);
199 break;
200 case "playableOnPreview":
201 properties.playableOnPreview = string.Equals(value, "true", StringComparison.OrdinalIgnoreCase);
202 break;
203 case "translationMod":
204 properties.translationMod = string.Equals(value, "true", StringComparison.OrdinalIgnoreCase);
205 break;
206 case "hideCode":
207 properties.hideCode = string.Equals(value, "true", StringComparison.OrdinalIgnoreCase);
208 break;
209 case "hideResources":
210 properties.hideResources = string.Equals(value, "true", StringComparison.OrdinalIgnoreCase);
211 break;
212 case "includeSource":
213 properties.includeSource = string.Equals(value, "true", StringComparison.OrdinalIgnoreCase);
214 break;
215 case "buildIgnore":
216 properties.buildIgnores = (from s in value.Split(',')
217 select s.Trim().Replace('\\', Path.DirectorySeparatorChar).Replace('/', Path.DirectorySeparatorChar) into s
218 where s.Length > 0
219 select s).ToArray();
220 break;
221 case "side":
222 if (!Enum.TryParse<ModSide>(value, ignoreCase: true, out properties.side))
223 {
224 throw new Exception("side is not one of (Both, Client, Server, NoSync): " + value);
225 }
226 break;
227 }
228 }
229 List<string> refs = properties.RefNames(includeWeak: true).ToList();
230 if (refs.Count != refs.Distinct().Count())
231 {
232 throw new Exception("Duplicate mod/weak reference");
233 }
234 properties.sortAfter = (from dep in properties.RefNames(includeWeak: true)
235 where !properties.sortBefore.Contains(dep)
236 select dep).Concat(properties.sortAfter).Distinct().ToArray();
237 return properties;
238 }
