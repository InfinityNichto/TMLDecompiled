51 internal static LaunchResult Launch()
52 {

57 }

59 string tMLName = Path.GetFileName(Assembly.GetExecutingAssembly().Location);
60 Process proc = new Process
61 {
62 StartInfo =
63 {
64 FileName = Environment.ProcessPath,
65 Arguments = tMLName + " -terrariasteamclient " + serverPipe.GetClientHandleAsString(),
66 UseShellExecute = false,
67 RedirectStandardOutput = true
68 }
69 };
70 string[] array = ((IEnumerable<string>)proc.StartInfo.EnvironmentVariables.Keys).ToArray();
71 foreach (string i in array)
72 {
73 if (i.StartsWith("steam", StringComparison.InvariantCultureIgnoreCase))
74 {
75 proc.StartInfo.EnvironmentVariables.Remove(i);
76 }
77 }
78 proc.Start();
79 while (true)
80 {
81 string line = proc.StandardOutput.ReadLine()?.Trim();
82 if (line == null)
83 {
84 if (proc.HasExited)
85 {
86 return LaunchResult.ErrClientProcDied;
87 }
88 continue;
89 }
90 Logger.Debug((object)("Recv: " + line));
91 if (line == MsgInitFailed)
92 {
93 return LaunchResult.ErrSteamInitFailed;
94 }
95 if (line == MsgNotInstalled)
96 {
97 return LaunchResult.ErrNotInstalled;
98 }
99 if (line == MsgInstallOutOfDate)
100 {
101 return LaunchResult.ErrInstallOutOfDate;
102 }
103 if (line == MsgInitSuccess)
104 {
105 break;
106 }
107 if (line == MsgFamilyShared)
108 {
109 SteamedWraps.FamilyShared = true;
110 }
111 }
112 SendCmd(MsgAck);
113 Thread.Sleep(300);
114 return LaunchResult.Ok;
115 }
