298 string modNameTrimmed = _modName.CurrentString.Trim();
299 string basicSwordTrimmed = _basicSword.CurrentString.Trim();
300 string sourceFolder = Path.Combine(ModCompile.ModSourcePath, modNameTrimmed);
301 CodeDomProvider provider = CodeDomProvider.CreateProvider("C#");
302 if (Directory.Exists(sourceFolder))
303 {
304 _messagePanel.SetText("Folder already exists");
305 return;
306 }
307 if (!provider.IsValidIdentifier(modNameTrimmed))
308 {
309 _messagePanel.SetText("ModName is invalid C# identifier. Remove spaces.");
310 return;
311 }
312 if (modNameTrimmed.Equals("Mod", StringComparison.InvariantCultureIgnoreCase) || modNameTrimmed.Equals("ModLoader", StringComparison.InvariantCultureIgnoreCase) || modNameTrimmed.Equals("tModLoader", StringComparison.InvariantCultureIgnoreCase))
313 {
314 _messagePanel.SetText("ModName is a reserved mod name. Choose a different name.");
315 return;
316 }
317 if (!string.IsNullOrEmpty(basicSwordTrimmed) && !provider.IsValidIdentifier(basicSwordTrimmed))
318 {
319 _messagePanel.SetText("BasicSword is invalid C# identifier. Remove spaces.");
320 return;
321 }
322 if (string.IsNullOrWhiteSpace(_modDiplayName.CurrentString))
323 {
324 _messagePanel.SetText("DisplayName can't be empty");

326 }
327 if (string.IsNullOrWhiteSpace(_modAuthor.CurrentString))
328 {
329 _messagePanel.SetText("Author can't be empty");

331 }
332 Directory.CreateDirectory(sourceFolder);
333 File.WriteAllText(Path.Combine(sourceFolder, "build.txt"), GetModBuild());
334 File.WriteAllText(Path.Combine(sourceFolder, "description.txt"), GetModDescription());
335 using (Stream stream = Assembly.GetExecutingAssembly().GetManifestResourceStream("Terraria.ModLoader.Default.iconTemplate.png"))

342 string text = Path.Combine(sourceFolder, "Properties");
