99 private static DetourList GetDetourList(Assembly asm)
100 {
101 if (asm == typeof(Action).Assembly)
102 {
103 throw new ArgumentException("Cannot identify owning assembly of hook. Make sure there are no delegate type changing wrappers between the method/lambda and the Modify/Add/+= call. Eg `new ILContext.Manipulator(action)` is bad");
104 }
105 if (!assemblyDetours.TryGetValue(asm, out var list))
106 {
107 return assemblyDetours[asm] = new DetourList();
108 }
109 return list;
110 }

123 DetourManager.DetourApplied += delegate(DetourInfo info)
124 {
125 Assembly assembly2 = info.Entry.DeclaringType.Assembly;
126 GetDetourList(assembly2).detours.Add(info);
127 string text = "Hook " + StringRep(((DetourBase)info).Method.Method) + " added by " + assembly2.GetName().Name;
128 MethodSignature val = MethodSignature.ForMethod(((DetourBase)info).Method.Method);
129 MethodSignature val2 = MethodSignature.ForMethod(info.Entry, true);
130 if (val2.ParameterCount != val.ParameterCount + 1 || (object)val2.FirstParameter.GetMethod("Invoke") == null)
131 {
132 text += " WARNING! No orig delegate, incompatible with other hooks to this method";
133 }
134 Logging.tML.Debug((object)text);
135 };
136 DetourManager.ILHookApplied += delegate(ILHookInfo info)
137 {
138 Assembly assembly = info.ManipulatorMethod.DeclaringType.Assembly;
139 GetDetourList(assembly).ilHooks.Add(info);
140 Logging.tML.Debug((object)("ILHook " + StringRep(((DetourBase)info).Method.Method) + " added by " + assembly.GetName().Name));
141 };
142 isInitialized = true;
143 }

145 private static string StringRep(MethodBase m)
146 {
147 string paramString = string.Join(", ", m.GetParameters().Select(delegate(ParameterInfo p)
148 {
149 Type type = p.ParameterType;
150 string text = "";
151 if (type.IsByRef)
152 {
153 text = (p.IsOut ? "out " : "ref ");
154 type = type.GetElementType();
155 }
156 string value;
157 return text + (defaultAliases.TryGetValue(type, out value) ? value : type.Name);
158 }));
159 string owner = m.DeclaringType?.FullName ?? ((m is DynamicMethod) ? "dynamic" : "unknown");
160 return $"{owner}::{m.Name}({paramString})";
161 }

163 internal static void RemoveAll(Mod mod)
164 {
165 if (mod is ModLoaderMod)
166 {
167 return;
168 }
169 foreach (Assembly asm in AssemblyManager.GetModAssemblies(mod.Name))
170 {
171 if (!assemblyDetours.TryGetValue(asm, out var list))
172 {
173 continue;
174 }
175 Logging.tML.Debug((object)$"Unloading {list.ilHooks.Count} IL hooks, {list.detours.Count} detours from {asm.GetName().Name} in {mod.DisplayName}");
176 foreach (DetourInfo detour in list.detours)
177 {
178 if (((DetourBase)detour).IsApplied)
179 {
180 ((DetourBase)detour).Undo();
181 }
182 }
183 foreach (ILHookInfo ilHook in list.ilHooks)
184 {
185 if (((DetourBase)ilHook).IsApplied)
186 {
187 ((DetourBase)ilHook).Undo();
188 }
189 }
190 }
191 }

273 public static void DumpIL(Mod mod, ILContext il)
274 {
275 string methodName = ((MemberReference)il.Method).FullName.Replace(':', '_');
276 if (methodName.Contains('?'))
277 {
278 string text = methodName;
279 int num = methodName.LastIndexOf('?') + 1;
280 methodName = text.Substring(num, text.Length - num);
281 }
282 string filePath = Path.Combine(Logging.LogDir, "ILDumps", mod.Name, methodName + ".txt");
283 string folderPath = Path.GetDirectoryName(filePath);
284 if (!Directory.Exists(folderPath))
285 {
286 Directory.CreateDirectory(folderPath);
287 }
288 File.WriteAllText(filePath, ((object)il).ToString());
289 Logging.tML.Debug((object)$"Dumped ILContext \"{((MemberReference)il.Method).FullName}\" to \"{filePath}\"");
