364 internal static void ServerModMenu(out bool reloadMods)
365 {
366 bool exit = false;
367 reloadMods = false;
368 while (!exit)
369 {
370 Console.WriteLine("Terraria Server " + Main.versionNumber2 + " - " + ModLoader.versionedName);
371 Console.WriteLine();
372 LocalMod[] mods = ModOrganizer.FindMods(logDuplicates: true);
373 for (int i = 0; i < mods.Length; i++)
374 {
375 Console.Write(i + 1 + "\t\t" + mods[i].DisplayName);
376 Console.WriteLine(" (" + (mods[i].Enabled ? "enabled" : "disabled") + ")");
377 }
378 if (mods.Length == 0)
379 {
380 Console.ForegroundColor = ConsoleColor.Yellow;
381 Console.WriteLine(Language.GetTextValue("tModLoader.ModsNotFoundServer", ModLoader.ModPath));
382 Console.ResetColor();
383 }
384 Console.WriteLine("e\t\t" + Language.GetTextValue("tModLoader.ModsEnableAll"));
385 Console.WriteLine("d\t\t" + Language.GetTextValue("tModLoader.ModsDisableAll"));
386 Console.WriteLine("r\t\t" + Language.GetTextValue("tModLoader.ModsReloadAndReturn"));
387 Console.WriteLine(Language.GetTextValue("tModLoader.AskForModIndex"));
388 Console.WriteLine();
389 Console.WriteLine(Language.GetTextValue("tModLoader.AskForCommand"));
390 string command = Console.ReadLine();
391 if (command == null)
392 {
393 command = "";
394 }
395 command = command.ToLower();
396 Console.Clear();
397 switch (command)
398 {
399 case "e":
400 {
401 LocalMod[] array = mods;
402 for (int j = 0; j < array.Length; j++)
403 {
404 array[j].Enabled = true;
405 }
406 continue;
407 }
408 case "d":
409 {
410 LocalMod[] array = mods;
411 for (int j = 0; j < array.Length; j++)
412 {
413 array[j].Enabled = false;
414 }
415 continue;
416 }
417 case "r":
418 reloadMods = true;
419 exit = true;
420 continue;
421 }
422 if (!int.TryParse(command, out var value) || value <= 0 || value > mods.Length)
423 {
424 continue;
425 }
426 LocalMod mod = mods[value - 1];
427 mod.Enabled = !mod.Enabled;
428 if (mod.Enabled)
429 {
430 List<string> missingRefs = new List<string>();
431 EnableDepsRecursive(mod, mods, missingRefs);
432 if (missingRefs.Any())
433 {
434 Console.ForegroundColor = ConsoleColor.Red;
435 Console.WriteLine(Language.GetTextValue("tModLoader.ModDependencyModsNotFound", string.Join(", ", missingRefs)) + "\n");
436 Console.ResetColor();
437 }
438 }
439 }

442 private static void EnableDepsRecursive(LocalMod mod, LocalMod[] mods, List<string> missingRefs)
443 {
444 string[] array = mod.properties.modReferences.Select((BuildProperties.ModReference x) => x.mod).ToArray();
445 foreach (string name in array)
446 {
447 LocalMod dep = mods.FirstOrDefault((LocalMod x) => x.Name == name);
448 if (dep == null)
449 {
450 missingRefs.Add(name);
451 continue;
452 }
453 EnableDepsRecursive(dep, mods, missingRefs);
454 if (!dep.Enabled)
455 {
456 Console.ForegroundColor = ConsoleColor.Green;
457 Console.WriteLine("Automatically enabling " + dep.DisplayName + " required by " + mod.DisplayName);
458 Console.ResetColor();
459 }
460 dep.Enabled = !dep.Enabled;
461 }
