33 public BestiaryEntry Register(BestiaryEntry entry)
34 {
35 _entries.Add(entry);
36 for (int i = 0; i < entry.Info.Count; i++)
37 {
38 if (entry.Info[i] is NPCNetIdBestiaryInfoElement nPCNetIdBestiaryInfoElement)
39 {
40 _byNpcId[nPCNetIdBestiaryInfoElement.NetId] = entry;
41 }
42 }
43 Mod mod = ContentSamples.NpcsByNetId[((NPCNetIdBestiaryInfoElement)entry.Info[0]).NetId].ModNPC?.Mod;
44 if (mod == null)
45 {
46 _vanillaEntries.Add(entry);
47 }
48 else if (_byMod.ContainsKey(mod))
49 {
50 _byMod[mod].Add(entry);
51 }
52 else
53 {
54 _byMod.Add(mod, new List<BestiaryEntry> { entry });

89 private void ExtractDropsForNPC(ItemDropDatabase dropsDatabase, int npcId)
90 {
91 BestiaryEntry bestiaryEntry = FindEntryByNPCID(npcId);
92 if (bestiaryEntry == null)
93 {
94 return;
95 }
96 List<IItemDropRule> rulesForNPCID = dropsDatabase.GetRulesForNPCID(npcId, includeGlobalDrops: false);
97 List<DropRateInfo> list = new List<DropRateInfo>();
98 DropRateInfoChainFeed ratesInfo = new DropRateInfoChainFeed(1f);
99 foreach (IItemDropRule item3 in rulesForNPCID)
100 {
101 item3.ReportDroprates(list, ratesInfo);
102 }
103 foreach (DropRateInfo item2 in list)
104 {
105 bestiaryEntry.Info.Add(new ItemDropBestiaryInfoElement(item2));
106 }
107 }
