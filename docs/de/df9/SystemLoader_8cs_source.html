355 public static void ModifyLightingBrightness(ref float negLight, ref float negLight2)
356 {
357 float scale = 1f;
358 ReadOnlySpan<ModSystem> readOnlySpan = HookModifyLightingBrightness.Enumerate();
359 for (int i = 0; i < readOnlySpan.Length; i++)
360 {
361 readOnlySpan[i].ModifyLightingBrightness(ref scale);
362 }
363 if (Lighting.NotRetro)
364 {
365 negLight *= scale;
366 negLight2 *= scale;
367 }
368 else
369 {
370 negLight -= (scale - 1f) / 2.3076923f;
371 negLight2 -= (scale - 1f) / 0.75f;
372 }
373 negLight = Math.Max(negLight, 0.001f);
374 negLight2 = Math.Max(negLight2, 0.001f);
375 }
