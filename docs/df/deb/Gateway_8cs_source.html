100 private static string GetLocation(string data)
101 {
102 foreach (string line in from l in data.Split('\n')
103 select l.Trim() into l
104 where l.Length > 0
105 select l)
106 {
107 if (line.StartsWith("HTTP/1.") || line.StartsWith("NOTIFY *"))
108 {
109 continue;
110 }
111 int colonIndex = line.IndexOf(':');
112 if (colonIndex < 0)
113 {
114 continue;
115 }
116 string name = line.Substring(0, colonIndex);
117 object obj;
118 if (line.Length < name.Length)
119 {
120 obj = null;
121 }
122 else
123 {
124 string text = line;
125 int num = colonIndex + 1;
126 obj = text.Substring(num, text.Length - num).Trim();
127 }
128 string val = (string)obj;
129 if (name.ToLowerInvariant() == "location")
130 {
131 if (val.IndexOf('/', 7) == -1)
132 {
133 throw new Exception("Unsupported Gateway");
134 }
135 return val;
136 }
137 }
138 throw new Exception("Unsupported Gateway");
139 }

141 private static (string serviceType, string controlURL) GetInfo(string location)
142 {
143 IEnumerable<XElement> enumerable = from d in XDocument.Load(location).Descendants()
144 where d.Name.LocalName == "service"
145 select d;
146 (string, string) ret = (null, null);
147 foreach (XElement item in enumerable)
148 {
149 string serviceType = null;
150 string controlURL = null;
151 foreach (XNode item2 in item.Nodes())
152 {
153 if (!(item2 is XElement { FirstNode: XText i } ele))
154 {
155 continue;
156 }
157 string text = ele.Name.LocalName.Trim().ToLowerInvariant();
158 if (!(text == "servicetype"))
159 {
160 if (text == "controlurl")
161 {
162 controlURL = i.Value.Trim();
163 }
164 }
165 else
166 {
167 serviceType = i.Value.Trim();
168 }
169 }
170 if (serviceType != null && controlURL != null && (serviceType.ToLowerInvariant().Contains(":wanipconnection:") || serviceType.ToLowerInvariant().Contains(":wanpppconnection:")))
171 {
172 ret.Item1 = serviceType;
173 ret.Item2 = controlURL;
174 }
175 }
176 if (ret.Item2 == null)
177 {
178 throw new Exception("Unsupported Gateway");
179 }
180 if (!ret.Item2.StartsWith('/'))
181 {
182 ret.Item2 = "/" + ret.Item2;
183 }
184 ret.Item2 = location[..location.IndexOf('/', 7)] + ret.Item2;
