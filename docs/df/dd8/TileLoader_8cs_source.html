313 public static void CheckModTile(int i, int j, int type)
314 {
315 if (type <= TileID.Count || WorldGen.destroyObject)
316 {
317 return;
318 }
319 TileObjectData tileData = TileObjectData.GetTileData(type, 0);
320 if (tileData == null)
321 {
322 return;
323 }
324 int frameX = Main.tile[i, j].frameX;
325 int frameY = Main.tile[i, j].frameY;
326 int subX = frameX / tileData.CoordinateFullWidth;
327 int subY = frameY / tileData.CoordinateFullHeight;
328 int wrap = tileData.StyleWrapLimit;
329 if (wrap == 0)
330 {
331 wrap = 1;
332 }
333 int num = (tileData.StyleHorizontal ? (subY * wrap + subX) : (subX * wrap + subY));
334 int style = num / tileData.StyleMultiplier;
335 int alternate = num % tileData.StyleMultiplier;
336 for (int k = 0; k < tileData.AlternatesCount; k++)
337 {
338 if (alternate >= tileData.Alternates[k].Style && alternate <= tileData.Alternates[k].Style + tileData.RandomStyleRange)
339 {
340 alternate = k;
341 break;
342 }
343 }
344 tileData = TileObjectData.GetTileData(type, style, alternate + 1);
345 int partFrameX = frameX % tileData.CoordinateFullWidth;
346 int partFrameY = frameY % tileData.CoordinateFullHeight;
347 int partX = partFrameX / (tileData.CoordinateWidth + tileData.CoordinatePadding);
348 int partY = 0;
349 for (int remainingFrameY = partFrameY; partY < tileData.Height && remainingFrameY - tileData.CoordinateHeights[partY] + tileData.CoordinatePadding >= 0; partY++)
350 {
351 remainingFrameY -= tileData.CoordinateHeights[partY] + tileData.CoordinatePadding;
352 }
353 int originalI = i;
354 int originalJ = j;
355 i -= partX;
356 j -= partY;
357 int originX = i + tileData.Origin.X;
358 int originY = j + tileData.Origin.Y;
359 bool partiallyDestroyed = false;
360 for (int x3 = i; x3 < i + tileData.Width; x3++)
361 {
362 for (int y3 = j; y3 < j + tileData.Height; y3++)
363 {
364 if (!Main.tile[x3, y3].active() || Main.tile[x3, y3].type != type)
365 {
366 partiallyDestroyed = true;
367 break;
368 }
369 }
370 if (partiallyDestroyed)
371 {
372 break;
373 }
374 }
375 if (partiallyDestroyed || !TileObject.CanPlace(originX, originY, type, style, 0, out var _, onlyCheck: true, null, checkStay: true))
376 {
377 WorldGen.destroyObject = true;
378 if (tileData.Width != 1 || tileData.Height != 1)
379 {
380 WorldGen.KillTile_DropItems(originalI, originalJ, Main.tile[originalI, originalJ], includeLargeObjectDrops: true, includeAllModdedLargeObjectDrops: true);
381 }
382 for (int x2 = i; x2 < i + tileData.Width; x2++)
383 {
384 for (int y2 = j; y2 < j + tileData.Height; y2++)
385 {
386 if (Main.tile[x2, y2].type == type && Main.tile[x2, y2].active())
387 {
388 WorldGen.KillTile(x2, y2);
389 }
390 }
391 }
392 KillMultiTile(i, j, frameX - partFrameX, frameY - partFrameY, type);
393 WorldGen.destroyObject = false;
394 for (int x = i - 1; x < i + tileData.Width + 2; x++)
395 {
396 for (int y = j - 1; y < j + tileData.Height + 2; y++)
397 {
398 WorldGen.TileFrame(x, y);
399 }
400 }
401 }
402 TileObject.objectPreview.Active = false;
403 }
