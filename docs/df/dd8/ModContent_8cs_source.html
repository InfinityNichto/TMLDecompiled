84 public static void SplitName(string name, out string domain, out string subName)
85 {
86 int slash = name.IndexOfAny(nameSplitters);
87 if (slash < 0)
88 {
89 throw new MissingResourceException(Language.GetTextValue("tModLoader.LoadErrorMissingModQualifier", name));
90 }
91 domain = name.Substring(0, slash);
92 subName = name.Substring(slash + 1);
93 }

131 public static Asset<T> Request<T>(string name, AssetRequestMode mode = AssetRequestMode.AsyncLoad) where T : class
132 {
133 SplitName(name, out var modName, out var subName);
134 if (Main.dedServ && Main.Assets == null)
135 {
136 Main.Assets = new AssetRepository(null);
137 }
138 if (modName == "Terraria")
139 {
140 return Main.Assets.Request<T>(subName, mode);
141 }
142 if (!ModLoader.TryGetMod(modName, out var mod))
143 {
144 throw new MissingResourceException(Language.GetTextValue("tModLoader.LoadErrorModNotFoundDuringAsset", modName, name));
145 }
146 return mod.Assets.Request<T>(subName, mode);

153 public static bool HasAsset(string name)
154 {
155 if (Main.dedServ || string.IsNullOrWhiteSpace(name) || !name.Contains('/'))
156 {
157 return false;
158 }
159 SplitName(name, out var modName, out var subName);
160 if (modName == "Terraria")
161 {
162 return Main.AssetSourceController.StaticSource.HasAsset(subName);
163 }
164 if (ModLoader.TryGetMod(modName, out var mod))
165 {
166 return mod.RootContentSource.HasAsset(subName);
167 }
168 return false;
169 }

482 private static void LoadModContent(CancellationToken token, Action<Mod> loadAction)
483 {
484 MemoryTracking.Checkpoint();
485 int num = 0;
486 Mod[] mods = ModLoader.Mods;
487 foreach (Mod mod in mods)
488 {
489 token.ThrowIfCancellationRequested();
490 Interface.loadMods.SetCurrentMod(num++, mod);
491 try
492 {
493 loadAction(mod);
494 }
495 catch (Exception ex)
496 {
497 ex.Data["mod"] = mod.Name;
498 throw;
499 }
500 finally
501 {
502 MemoryTracking.Update(mod.Name);
503 }
504 }
505 }
