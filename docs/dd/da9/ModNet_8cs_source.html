394 internal static void ReceiveMod(BinaryReader reader)
395 {
396 if (downloadingMod == null)
397 {
398 return;
399 }
400 try
401 {
402 if (downloadingFile == null)
403 {
404 Interface.progress.Show(reader.ReadString(), 0, CancelDownload);
405 if (ModLoader.TryGetMod(downloadingMod.name, out var mod))
406 {
407 mod.Close();
408 }
409 downloadingLength = reader.ReadInt64();
410 Logging.tML.Debug((object)$"Downloading: {downloadingMod.name} {downloadingLength}bytes");
411 downloadingFile = new FileStream(downloadingMod.path, FileMode.Create);
412 return;
413 }
414 byte[] bytes = reader.ReadBytes((int)Math.Min(downloadingLength - downloadingFile.Position, 16384L));
415 downloadingFile.Write(bytes, 0, bytes.Length);
416 Interface.progress.Progress = (float)downloadingFile.Position / (float)downloadingLength;
417 if (downloadingFile.Position == downloadingLength)
418 {
419 downloadingFile.Close();
420 TmodFile mod2 = new TmodFile(downloadingMod.path);
421 using (mod2.Open())
422 {
423 }
424 if (!downloadingMod.Matches(mod2))
425 {
426 throw new Exception(Language.GetTextValue("tModLoader.MPErrorModHashMismatch"));
427 }
428 if (downloadingMod.signed && onlyDownloadSignedMods && !mod2.ValidModBrowserSignature)
429 {
430 throw new Exception(Language.GetTextValue("tModLoader.MPErrorModNotSigned"));
431 }
432 ModLoader.EnableMod(mod2.Name);
433 if (downloadQueue.Count > 0)
434 {
435 DownloadNextMod();
436 }
437 else
438 {
439 OnModsDownloaded(needsReload: true);
440 }
441 }
442 }
443 catch (Exception e)
444 {
445 try
446 {
447 downloadingFile?.Close();
448 File.Delete(downloadingMod.path);
449 }
450 catch (Exception exc2)
451 {
452 Logging.tML.Error((object)"Unknown error during mod sync", exc2);
453 }
454 string msg = Language.GetTextValue("tModLoader.MPErrorModDownloadError", downloadingMod.name);
455 Logging.tML.Error((object)msg, e);
456 Interface.errorMessage.Show(msg + e, 0);
457 Netplay.Disconnect = true;
458 downloadingMod = null;
459 }

561 {
562 if (netMods == null)

564 ReadNetIDs(reader);
565 return;
566 }
567 short id = ((NetModCount < 256) ? reader.ReadByte() : reader.ReadInt16());
568 int start = (int)reader.BaseStream.Position;

571 {
572 ReadUnderflowBypass = false;

574 if (!ReadUnderflowBypass && reader.BaseStream.Position - start != actualLength)

576 throw new IOException($"Read underflow {reader.BaseStream.Position - start} of {actualLength} bytes caused by {GetMod(id).Name} in HandlePacket");
