46 internal static bool WriteModMap(BinaryWriter writer)
47 {
48 ISet<ushort> types = new HashSet<ushort>();
49 for (int i = 0; i < Main.maxTilesX; i++)
50 {
51 for (int j = 0; j < Main.maxTilesY; j++)
52 {
53 ushort type2 = Main.Map[i, j].Type;
54 if (type2 >= MapHelper.modPosition)
55 {
56 types.Add(type2);
57 }
58 }
59 }
60 if (types.Count == 0)
61 {
62 return false;
63 }
64 writer.Write((ushort)types.Count);
65 foreach (ushort type in types)
66 {
67 writer.Write(type);
68 if (MapLoader.entryToTile.ContainsKey(type))
69 {
70 ModTile tile = TileLoader.GetTile(MapLoader.entryToTile[type]);
71 writer.Write(value: true);
72 writer.Write(tile.Mod.Name);
73 writer.Write(tile.Name);
74 writer.Write((ushort)(type - MapHelper.tileLookup[tile.Type]));
75 }
76 else if (MapLoader.entryToWall.ContainsKey(type))
77 {
78 ModWall wall = WallLoader.GetWall(MapLoader.entryToWall[type]);
79 writer.Write(value: false);
80 writer.Write(wall.Mod.Name);
81 writer.Write(wall.Name);
82 writer.Write((ushort)(type - MapHelper.wallLookup[wall.Type]));
83 }
84 else
85 {
86 writer.Write(value: true);
87 writer.Write("");
88 writer.Write("");
89 writer.Write((ushort)0);
90 }
91 }
92 WriteMapData(writer);
93 return true;

209 internal static void WriteMapTile(ref int i, ref int j, BinaryWriter writer, ref bool nextModTile)
210 {
211 MapTile tile = Main.Map[i, j];
212 byte flags = 0;
213 byte[] data = new byte[9];
214 int index = 1;
215 data[index] = (byte)tile.Type;
216 index++;
217 data[index] = (byte)(tile.Type >> 8);
218 index++;
219 if (tile.Light < byte.MaxValue)
220 {
221 flags = (byte)(flags | 1u);
222 data[index] = tile.Light;
223 index++;
224 }
225 if (tile.Color > 0)
226 {
227 flags = (byte)(flags | 2u);
228 data[index] = tile.Color;
229 index++;
230 }
231 int nextI = i;
232 int nextJ = j;
233 uint sameCount = 0u;
234 while (NextTile(ref nextI, ref nextJ))
235 {
236 MapTile nextTile = Main.Map[nextI, nextJ];
237 if (tile.Equals(ref nextTile) && sameCount < uint.MaxValue)
238 {
239 sameCount++;
240 i = nextI;
241 j = nextJ;
242 continue;
243 }
244 if (nextTile.Type >= MapHelper.modPosition && nextTile.Light > 18)
245 {
246 flags = (byte)(flags | 0x20u);
247 nextModTile = true;
248 }
249 break;
250 }
251 if (sameCount != 0)
252 {
253 flags = (byte)(flags | 4u);
254 data[index] = (byte)sameCount;
255 index++;
256 if (sameCount > 255)
257 {
258 flags = (byte)(flags | 8u);
259 data[index] = (byte)(sameCount >> 8);
260 index++;
261 if (sameCount > 65535)
262 {
263 flags = (byte)(flags | 0x10u);
264 data[index] = (byte)(sameCount >> 16);
265 index++;
266 data[index] = (byte)(sameCount >> 24);
267 index++;
268 }

275 internal static void ReadMapTile(ref int i, ref int j, IDictionary<ushort, ushort> table, BinaryReader reader, ref bool nextModTile)
276 {
277 byte flags = reader.ReadByte();
278 ushort type = table[reader.ReadUInt16()];
279 byte light = (((flags & 1) == 1) ? reader.ReadByte() : byte.MaxValue);
280 byte color = (byte)(((flags & 2) == 2) ? reader.ReadByte() : 0);
281 MapTile tile = MapTile.Create(type, light, color);
282 Main.Map.SetTile(i, j, ref tile);
283 if ((flags & 4) == 4)
284 {
285 uint sameCount = (((flags & 0x10) == 16) ? reader.ReadUInt32() : (((flags & 8) != 8) ? reader.ReadByte() : reader.ReadUInt16()));
286 for (uint k = 0u; k < sameCount; k++)
287 {
288 NextTile(ref i, ref j);
289 tile = MapTile.Create(type, light, color);
290 Main.Map.SetTile(i, j, ref tile);
291 }
292 }
293 if ((flags & 0x20) == 32)

295 nextModTile = true;
