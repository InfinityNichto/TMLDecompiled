247 internal void Check(NPCSpawnInfo info, ref float remainingWeight)
248 {
249 if (WeightFunc != null)
250 {
251 blockWeight = WeightFunc();
252 }
253 active = true;
254 if (!condition(info))
255 {
256 return;
257 }
258 chance = remainingWeight * blockWeight;
259 float childWeight = chance;
260 foreach (SpawnCondition child in children)
261 {
262 child.Check(info, ref childWeight);
263 if ((double)Math.Abs(childWeight) < 5E-06)
264 {
265 break;
266 }
267 }
268 remainingWeight -= chance;
269 }
