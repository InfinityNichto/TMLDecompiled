144 public static void Update(int buff, Player player, ref int buffIndex)
145 {
146 int originalIndex = buffIndex;
147 if (IsModBuff(buff))
148 {
149 GetBuff(buff).Update(player, ref buffIndex);
150 }
151 DelegateUpdatePlayer[] hookUpdatePlayer = HookUpdatePlayer;
152 foreach (DelegateUpdatePlayer hook in hookUpdatePlayer)
153 {
154 if (buffIndex == originalIndex)
155 {
156 hook(buff, player, ref buffIndex);
157 continue;
158 }
159 break;
160 }
161 }
