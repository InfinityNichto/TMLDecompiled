236 {
237 return Regex.Replace(name, "[^a-zA-Z0-9_.-]+", string.Empty, RegexOptions.Compiled);

240 internal static bool IsValidModpackName(string name)
241 {
242 if (!Regex.Match(name, "[^a-zA-Z0-9_.-]+", RegexOptions.Compiled).Success)
243 {
244 return name.IndexOfAny(Path.GetInvalidFileNameChars()) < 0;
245 }
246 return false;
247 }

256 public override void OnActivate()
257 {
258 _cts = new CancellationTokenSource();
259 _scrollPanel.Append(_uiLoader);
260 _modPacks.Clear();
261 Task.Run(delegate
262 {
263 Directory.CreateDirectory(ModPacksDirectory);
264 string[] directories = Directory.GetDirectories(ModPacksDirectory, "*", SearchOption.TopDirectoryOnly);
265 string[] files = Directory.GetFiles(ModPacksDirectory, "*.json", SearchOption.TopDirectoryOnly);
266 List<UIElement> ModPacksToAdd = new List<UIElement>();
267 foreach (string current in files.Concat(directories))
268 {
269 try
270 {
271 if (!IsValidModpackName(Path.GetFileNameWithoutExtension(current)))
272 {
273 throw new Exception();
274 }
275 if (Directory.Exists(current))
276 {
277 ModPacksToAdd.Add(LoadModernModPack(current));
278 }
279 else
280 {
281 ModPacksToAdd.Add(LoadLegacyModPack(current));
282 }
283 }
284 catch
285 {
286 UIAutoScaleTextTextPanel<string> item = new UIAutoScaleTextTextPanel<string>(Language.GetTextValue("tModLoader.ModPackMalformed", Path.GetFileName(current)))
287 {
288 Width =
289 {
290 Percent = 1f
291 },
292 Height =
293 {
294 Pixels = 50f,
295 Percent = 0f
296 }
297 };
298 ModPacksToAdd.Add(item);
299 }
300 }
301 Main.QueueMainThreadAction(delegate
302 {
303 _modPacks.AddRange(ModPacksToAdd);
304 _scrollPanel.RemoveChild(_uiLoader);
305 });
306 });
307 }

328 public static void SaveSnapshot(string configsPath, string modsPath)
329 {
330 if (!Directory.Exists(ConfigManager.ModConfigPath))
331 {
332 Directory.CreateDirectory(ConfigManager.ModConfigPath);
333 }
334 Directory.CreateDirectory(configsPath);
335 Directory.CreateDirectory(modsPath);
336 Directory.EnumerateFiles(ConfigManager.ModConfigPath);
337 File.Copy(Path.Combine(ModOrganizer.modPath, "enabled.json"), Path.Combine(modsPath, "enabled.json"), overwrite: true);
338 File.WriteAllText(Path.Combine(modsPath, "tmlversion.txt"), BuildInfo.tMLVersion.ToString());
339 List<string> workshopIds = new List<string>();
340 Mod[] mods = ModLoader.Mods;
341 foreach (Mod mod in mods)
342 {
343 if (mod.File != null)
344 {
345 if (ModOrganizer.TryReadManifest(ModOrganizer.GetParentDir(mod.File.path), out var info))
346 {
347 workshopIds.Add(info.workshopEntryId.ToString());
348 }
349 if (mod.File.path != Path.Combine(modsPath, mod.Name + ".tmod"))
350 {
351 File.Copy(mod.File.path, Path.Combine(modsPath, mod.Name + ".tmod"), overwrite: true);
352 }
353 }
354 }
355 File.WriteAllLines(Path.Combine(modsPath, "install.txt"), workshopIds);
356 }

358 public static void ExportSnapshot(string modPackName)
359 {
360 string instancePath = Path.Combine(Directory.GetCurrentDirectory(), modPackName);
361 Directory.CreateDirectory(instancePath);
362 Directory.CreateDirectory(Path.Combine(instancePath, "SaveData"));
363 string sourcePath = ModPackModsPath(modPackName);
364 string configPath = ConfigManager.ModConfigPath;
365 FileUtilities.CopyFolder(sourcePath, Path.Combine(instancePath, "SaveData", "Mods"));
366 FileUtilities.CopyFolder(configPath, Path.Combine(instancePath, "SaveData", "ModConfigs"));
367 File.WriteAllText(Path.Combine(instancePath, "cli-argsConfig.txt"), "-tmlsavedirectory " + Path.Combine(instancePath, "SaveData") + "\n-steamworkshopfolder none");
368 Logging.tML.Info((object)("Exported instance of Frozen Mod Pack " + modPackName + " to " + instancePath));
369 Utils.OpenFolder(instancePath);
370 }
