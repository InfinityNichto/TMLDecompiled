347 public static bool CalculateVersionsData(string workshopPath, ref NameValueCollection buildData)
348 {
349 foreach (string item in Directory.EnumerateFiles(workshopPath, "*.tmod*", SearchOption.AllDirectories))
350 {
351 LocalMod mod = OpenModFile(item);
352 if (mod.tModLoaderVersion.MajorMinor() <= BuildInfo.tMLVersion.MajorMinor() && mod.properties.version >= new Version(buildData["version"]))
353 {
354 return false;
355 }
356 if (mod.tModLoaderVersion.MajorMinor() != BuildInfo.tMLVersion.MajorMinor())
357 {
358 buildData["versionsummary"] += $";{mod.tModLoaderVersion}:{mod.properties.version}";
359 }
360 }
361 return true;
