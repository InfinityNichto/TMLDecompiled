153 private static void Load(CancellationToken token = default(CancellationToken))
154 {
155 if (isLoading)
156 {
157 throw new Exception("Load called twice");
158 }
159 isLoading = true;
160 if (!Unload())
161 {
162 return;
163 }
164 LocalMod[] availableMods = ModOrganizer.FindMods(logDuplicates: true);
165 try
166 {
167 List<Mod> list = AssemblyManager.InstantiateMods(ModOrganizer.SelectAndSortMods(availableMods, token), token);
168 list.Insert(0, new ModLoaderMod());
169 Mods = list.ToArray();
170 Mod[] mods = Mods;
171 foreach (Mod mod2 in mods)
172 {
173 modsByName[mod2.Name] = mod2;
174 }
175 ModContent.Load(token);
176 if (OnSuccessfulLoad != null)
177 {
178 OnSuccessfulLoad();
179 }
180 else
181 {
182 Main.menuMode = 0;
183 }
184 }
185 catch when (token.IsCancellationRequested)
186 {
187 skipLoad = true;
188 OnSuccessfulLoad = (Action)Delegate.Combine(OnSuccessfulLoad, (Action)delegate
189 {
190 Main.menuMode = 10000;
191 });
192 isLoading = false;
193 Load();
194 }
195 catch (Exception e)
196 {
197 List<string> responsibleMods = new List<string>();
198 if (e.Data.Contains("mod"))
199 {
200 responsibleMods.Add((string)e.Data["mod"]);
201 }
202 if (e.Data.Contains("mods"))
203 {
204 responsibleMods.AddRange((IEnumerable<string>)e.Data["mods"]);
205 }
206 responsibleMods.Remove("ModLoader");
207 if (responsibleMods.Count == 0 && AssemblyManager.FirstModInStackTrace(new StackTrace(e), out var stackMod))
208 {
209 responsibleMods.Add(stackMod);
210 }
211 string msg = Language.GetTextValue("tModLoader.LoadError", string.Join(", ", responsibleMods));
212 if (responsibleMods.Count == 1)
213 {
214 LocalMod mod = availableMods.FirstOrDefault((LocalMod m) => m.Name == responsibleMods[0]);
215 if (mod != null)
216 {
217 msg += $" v{mod.properties.version}";
218 }
219 if (mod != null && mod.tModLoaderVersion.MajorMinorBuild() != BuildInfo.tMLVersion.MajorMinorBuild())
220 {
221 msg = msg + "\n" + Language.GetTextValue("tModLoader.LoadErrorVersionMessage", mod.tModLoaderVersion, versionedName);
222 }
223 else if (mod != null)
224 {
225 SteamedWraps.QueueForceValidateSteamInstall();
226 }
227 if (e is JITException)
228 {
229 msg = msg + "\nThe mod will need to be updated to match the current tModLoader version, or may be incompatible with the version of some of your other mods. Click the '" + Language.GetTextValue("tModLoader.OpenWebHelp") + "' button to learn more.";
230 }
231 }
232 msg = ((responsibleMods.Count <= 0) ? (msg + "\n" + Language.GetTextValue("tModLoader.LoadErrorCulpritUnknown")) : (msg + "\n" + Language.GetTextValue("tModLoader.LoadErrorDisabled")));
233 if (e is ReflectionTypeLoadException reflectionTypeLoadException)
234 {
235 msg = msg + "\n\n" + string.Join("\n", reflectionTypeLoadException.LoaderExceptions.Select((Exception x) => x.Message));
236 }
237 if (e.Data.Contains("contentType") && e.Data["contentType"] is Type contentType)
238 {
239 msg = msg + "\n" + Language.GetTextValue("tModLoader.LoadErrorContentType", contentType.FullName);
240 }
241 Logging.tML.Error((object)msg, e);
242 foreach (string item in responsibleMods)
243 {
244 DisableMod(item);
245 }
246 isLoading = false;
247 DisplayLoadError(msg, e, e.Data.Contains("fatal"), responsibleMods.Count == 0);
248 }
249 finally
250 {
251 isLoading = false;
252 OnSuccessfulLoad = null;
253 skipLoad = false;
254 ModNet.NetReloadActive = false;
255 }
256 }

270 internal static bool Unload()
271 {
272 try
273 {
274 WeakReference<Mod>[] weakModRefs = GetWeakModRefs();
275 Mods_Unload();
276 WarnModsStillLoaded(weakModRefs);
277 return true;
278 }
279 catch (Exception e)
280 {
281 string msg = Language.GetTextValue("tModLoader.UnloadError");
282 if (e.Data.Contains("mod"))
283 {
284 msg = msg + "\n" + Language.GetTextValue("tModLoader.DefensiveUnload", e.Data["mod"]);
285 }
286 Logging.tML.Fatal((object)msg, e);
287 DisplayLoadError(msg, e, fatal: true);
288 return false;
289 }

318 private static void WarnModsStillLoaded(IReadOnlyList<WeakReference<Mod>> weakModRefs)
319 {
320 foreach (string alcName in AssemblyManager.OldLoadContexts().Distinct())
321 {
322 if (weakModRefs.Any((WeakReference<Mod> modRef) => modRef.TryGetTarget(out var target) && target.Name == alcName))
323 {
324 Logging.tML.WarnFormat(alcName + " mod class still using memory. Some content references have probably not been cleared. Use a heap dump to figure out why.", Array.Empty<object>());
325 }
326 else
327 {
328 Logging.tML.WarnFormat(alcName + " AssemblyLoadContext still using memory. Some classes are being held by Terraria or another mod. Use a heap dump to figure out why.", Array.Empty<object>());
329 }
330 }

358 public static bool IsSignedBy(TmodFile mod, string xmlPublicKey)
359 {
360 RSAPKCS1SignatureDeformatter rSAPKCS1SignatureDeformatter = new RSAPKCS1SignatureDeformatter();
361 AsymmetricAlgorithm v = AsymmetricAlgorithm.Create("RSA");
362 rSAPKCS1SignatureDeformatter.SetHashAlgorithm("SHA1");
363 v.FromXmlString(xmlPublicKey);
364 rSAPKCS1SignatureDeformatter.SetKey(v);
365 return rSAPKCS1SignatureDeformatter.VerifySignature(mod.Hash, mod.Signature);
366 }
