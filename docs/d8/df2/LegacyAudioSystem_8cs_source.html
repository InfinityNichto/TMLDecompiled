79 private IAudioTrack FindReplacementTrack(List<IContentSource> sources, string assetPath)
80 {
81 IAudioTrack audioTrack = null;
82 for (int i = 0; i < sources.Count; i++)
83 {
84 IContentSource contentSource = sources[i];
85 if (!contentSource.HasAsset(assetPath))
86 {
87 continue;
88 }
89 string extension = contentSource.GetExtension(assetPath);
90 string assetPathWithExtension = assetPath + extension;
91 try
92 {
93 IAudioTrack audioTrack2 = null;
94 switch (extension)
95 {
96 case ".ogg":
97 audioTrack2 = new OGGAudioTrack(contentSource.OpenStream(assetPathWithExtension));
98 break;
99 case ".wav":
100 audioTrack2 = new WAVAudioTrack(contentSource.OpenStream(assetPathWithExtension));
101 break;
102 case ".mp3":
103 audioTrack2 = new MP3AudioTrack(contentSource.OpenStream(assetPathWithExtension));
104 break;
105 }
106 if (audioTrack2 != null)
107 {
108 audioTrack?.Dispose();
109 audioTrack = audioTrack2;
110 }
111 }
112 catch
113 {
114 string textToShow = "A resource pack failed to load " + assetPath + "!";
115 Main.IssueReporter.AddReport(textToShow);
116 Main.IssueReporterIndicator.AttemptLettingPlayerKnow();
117 }
118 }
119 return audioTrack;
120 }
