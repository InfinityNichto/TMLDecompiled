70 return null;

90 public static void AddMusic(Mod mod, string musicPath)
91 {
92 if (!mod.loading)
93 {
94 throw new Exception("AddMusic can only be called during mod loading.");
95 }
96 int id = ReserveMusicID();
97 string chosenExtension = "";
98 foreach (string item in supportedExtensions.Where((string extension) => mod.FileExists(musicPath + extension)))
99 {
100 chosenExtension = item;
101 }
102 if (string.IsNullOrEmpty(chosenExtension))
103 {
104 throw new ArgumentException("Given path found no files matching the extensions [ " + string.Join(", ", supportedExtensions) + " ]");
105 }
106 musicPath = mod.Name + "/" + musicPath;
107 musicByPath[musicPath] = id;
108 musicExtensions[musicPath] = chosenExtension;

142 public static void AddMusicBox(Mod mod, int musicSlot, int itemType, int tileType, int tileFrameY = 0)
143 {
144 if (musicSlot < Main.maxMusic && (!Main.dedServ || musicSlot != 0))
145 {
146 throw new ArgumentOutOfRangeException($"Cannot assign music box to vanilla music ID {musicSlot}");
147 }
148 if (musicSlot >= MusicCount)
149 {
150 throw new ArgumentOutOfRangeException($"Music ID {musicSlot} does not exist");
151 }
152 if (itemType < ItemID.Count)
153 {
154 throw new ArgumentOutOfRangeException("Cannot assign music box to vanilla item ID " + itemType);
155 }
156 if (ItemLoader.GetItem(itemType) == null)
157 {
158 throw new ArgumentOutOfRangeException($"Item ID {itemType} does not exist");
159 }
160 if (tileType < TileID.Count)
161 {
162 throw new ArgumentOutOfRangeException($"Cannot assign music box to vanilla tile ID {tileType}");
163 }
164 if (TileLoader.GetTile(tileType) == null)
165 {
166 throw new ArgumentOutOfRangeException($"Tile ID {tileType} does not exist");
167 }
168 if (musicToItem.ContainsKey(musicSlot))
169 {
170 throw new ArgumentException($"Music ID {musicSlot} has already been assigned a music box");
171 }
172 if (itemToMusic.ContainsKey(itemType))
173 {
174 throw new ArgumentException($"Item ID {itemType} has already been assigned a music");
175 }
176 if (!tileToMusic.TryGetValue(tileType, out var tileToMusicDictionary))
177 {
178 tileToMusicDictionary = (tileToMusic[tileType] = new Dictionary<int, int>());
179 }
180 if (tileToMusicDictionary.ContainsKey(tileFrameY))
181 {
182 throw new ArgumentException($"Y-frame {tileFrameY} of tile type {tileType} has already been assigned a music");
183 }
184 if (tileFrameY % 36 != 0)
185 {
186 throw new ArgumentException("Y-frame must be divisible by 36");
187 }
188 if (!Main.dedServ)
189 {
190 musicToItem[musicSlot] = itemType;
191 itemToMusic[itemType] = musicSlot;
192 }
193 tileToMusic[tileType][tileFrameY] = musicSlot;
194 }

196 internal static void AutoloadMusic(Mod mod)
197 {
198 if (mod.File == null)

200 return;
201 }
202 foreach (string musicPath in from path in mod.RootContentSource.EnumerateAssets()
203 where supportedExtensions.Contains<string>(Path.GetExtension(path)) && (path.StartsWith("Music/") || path.Contains("/Music/"))

205 {

215 internal static IAudioTrack LoadMusic(string path, string extension)
216 {
217 path = "tmod:" + path + extension;
218 Stream stream = ModContent.OpenRead(path, newFileStream: true);
219 return extension switch
220 {
221 ".wav" => new WAVAudioTrack(stream),
222 ".mp3" => new MP3AudioTrack(stream),
223 ".ogg" => new OGGAudioTrack(stream),
224 _ => throw new ResourceLoadException("Unknown music extension " + extension),
225 };

257 void ILoader.ResizeArrays()
258 {
259 if (!(Main.audioSystem is LegacyAudioSystem legacyAudioSystem))
260 {
261 return;
262 }
263 Array.Resize(ref legacyAudioSystem.AudioTracks, MusicCount);
264 Array.Resize(ref Main.musicFade, MusicCount);
265 Array.Resize(ref Main.musicNoCrossFade, MusicCount);
266 foreach (string sound in musicByPath.Keys)
267 {
268 int slot = GetMusicSlot(sound);
269 if (Main.audioSystem is DisabledAudioSystem)
270 {
271 return;
272 }
273 legacyAudioSystem.AudioTracks[slot] = GetMusic(sound);
274 }
275 Main.audioSystem = legacyAudioSystem;
276 }
