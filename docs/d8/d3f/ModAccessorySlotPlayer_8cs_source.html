161 {
162 if (!slots.ContainsKey(name))

164 slots.Add(name, slots.Count);

167 ResetAndSizeAccessoryArrays();
168 List<Item> items = tag.GetList<TagCompound>("items").Select(ItemIO.Load).ToList();
169 List<Item> dyes = tag.GetList<TagCompound>("dyes").Select(ItemIO.Load).ToList();
170 List<bool> visible = tag.GetList<bool>("visible").ToList();
171 for (int i = 0; i < order.Count; i++)
172 {
173 int type = slots[order[i]];
174 exDyesAccessory[type] = dyes[i];
175 exHideAccessory[type] = visible[i];
176 exAccessorySlot[type] = items[i];

296 {

299 {

302 NetHandler.SendSlot(-1, base.Player.whoAmI, i, exAccessorySlot[i]);
303 }
304 if (exAccessorySlot[i + SlotCount].IsNetStateDifferent(clientInv.exAccessorySlot[i + LoadedSlotCount]))
305 {
306 NetHandler.SendSlot(-1, base.Player.whoAmI, i + LoadedSlotCount, exAccessorySlot[i + SlotCount]);
307 }
308 if (exDyesAccessory[i].IsNetStateDifferent(clientInv.exDyesAccessory[i]))
309 {
310 NetHandler.SendSlot(-1, base.Player.whoAmI, -i - 1, exDyesAccessory[i]);
311 }
312 if (exHideAccessory[i] != clientInv.exHideAccessory[i])
313 {
314 NetHandler.SendVisualState(-1, base.Player.whoAmI, i, exHideAccessory[i]);
