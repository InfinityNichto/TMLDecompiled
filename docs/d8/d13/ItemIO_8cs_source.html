53 public static TagCompound Save(Item item, List<TagCompound> globalData)
54 {
55 TagCompound tag = new TagCompound();
56 if (item.type <= 0)
57 {
58 return tag;
59 }
60 if (item.ModItem == null)
61 {
62 tag.Set("mod", "Terraria");
63 tag.Set("id", item.netID);
64 }
65 else
66 {
67 tag.Set("mod", item.ModItem.Mod.Name);
68 tag.Set("name", item.ModItem.Name);

71 if (saveData.Count > 0)
72 {
73 tag.Set("data", saveData);
74 }
75 }

77 if (modPrefix != null)

79 tag.Set("modPrefixMod", modPrefix.Mod.Name);
80 tag.Set("modPrefixName", modPrefix.Name);
81 }
82 else if (item.prefix != 0 && item.prefix < PrefixID.Count)
83 {
84 tag.Set("prefix", (byte)item.prefix);
85 }
86 if (item.stack > 1)
87 {
88 tag.Set("stack", item.stack);

90 if (item.favorited)

92 tag.Set("fav", true);

98 public static void Load(Item item, TagCompound tag)
99 {
100 string modName = tag.GetString("mod");
101 if (modName == "")
102 {
103 item.netDefaults(0);
104 return;
105 }
106 ModItem modItem;
107 if (modName == "Terraria")
108 {
109 item.netDefaults(tag.GetInt("id"));
110 }
111 else if (ModContent.TryFind<ModItem>(modName, tag.GetString("name"), out modItem))
112 {
113 item.SetDefaults(modItem.Type);
114 item.ModItem.LoadData(tag.GetCompound("data"));
115 }
116 else
117 {
118 item.SetDefaults(ModContent.ItemType<UnloadedItem>());
119 ((UnloadedItem)item.ModItem).Setup(tag);
120 }
121 if (tag.ContainsKey("modPrefixMod") && tag.ContainsKey("modPrefixName"))
122 {
123 item.Prefix(ModContent.TryFind<ModPrefix>(tag.GetString("modPrefixMod"), tag.GetString("modPrefixName"), out var prefix) ? prefix.Type : 0);
124 }
125 else if (tag.ContainsKey("prefix"))
126 {
127 item.Prefix(tag.GetByte("prefix"));
128 }
129 item.stack = tag.Get<int?>("stack") ?? 1;
130 item.favorited = tag.GetBool("fav");
131 if (!(item.ModItem is UnloadedItem))
132 {
133 LoadGlobals(item, tag.GetList<TagCompound>("globalData"));
134 }
135 }

144 internal static List<TagCompound> SaveGlobals(Item item)
145 {
146 if (item.ModItem is UnloadedItem)
147 {
148 return null;
149 }
150 List<TagCompound> list = new List<TagCompound>();
151 TagCompound saveData = new TagCompound();
152 EntityGlobalsEnumerator<GlobalItem> enumerator = ItemLoader.HookSaveData.Enumerate(item).GetEnumerator();
153 while (enumerator.MoveNext())
154 {
155 GlobalItem g = enumerator.Current;
156 if (g is UnloadedGlobalItem unloadedGlobalItem)
157 {
158 list.AddRange(unloadedGlobalItem.data);
159 continue;
160 }
161 g.SaveData(item, saveData);
162 if (saveData.Count != 0)
163 {
164 list.Add(new TagCompound
165 {
166 ["mod"] = g.Mod.Name,
167 ["name"] = g.Name,
168 ["data"] = saveData
169 });
170 saveData = new TagCompound();
171 }
172 }
173 if (list.Count <= 0)
174 {
175 return null;
176 }
177 return list;
178 }

180 internal static void LoadGlobals(Item item, IList<TagCompound> list)
181 {
182 foreach (TagCompound tag in list)
183 {
184 if (ModContent.TryFind<GlobalItem>(tag.GetString("mod"), tag.GetString("name"), out var globalItemBase) && item.TryGetGlobalItem(globalItemBase, out var globalItem))
185 {
186 try
187 {
188 globalItem.LoadData(item, tag.GetCompound("data"));

190 catch (Exception e)

193 }
194 }

297 internal static byte[] LegacyModData(int type, BinaryReader reader, bool hasGlobalSaving = true)
298 {
299 using MemoryStream memoryStream = new MemoryStream();
300 using (BinaryWriter writer = new BinaryWriter(memoryStream))
301 {
302 if (type >= ItemID.Count)
303 {
304 ushort length2 = reader.ReadUInt16();
305 writer.Write(length2);
306 writer.Write(reader.ReadBytes(length2));
307 }
308 if (hasGlobalSaving)
309 {
310 ushort count = reader.ReadUInt16();
311 writer.Write(count);
312 for (int i = 0; i < count; i++)
313 {
314 writer.Write(reader.ReadString());
315 writer.Write(reader.ReadString());
316 ushort length = reader.ReadUInt16();
317 writer.Write(length);
318 writer.Write(reader.ReadBytes(length));
319 }
320 }
321 }
