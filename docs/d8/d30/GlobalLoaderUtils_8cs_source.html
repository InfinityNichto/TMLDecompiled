36 public static void SetDefaults(TEntity entity, ref TGlobal[] entityGlobals, Action<TEntity> setModEntityDefaults)
37 {
38 if (entity.Type == 0)
39 {
40 return;
41 }
42 int initialType = entity.Type;
43 entityGlobals = new TGlobal[GlobalList<TGlobal>.SlotsPerEntity];
44 if (!GlobalTypeLookups<TGlobal>.Initialized)
45 {
46 SetDefaultsBeforeLookupsAreBuilt(entity, entityGlobals, setModEntityDefaults);
47 return;
48 }
49 TGlobal[] array = SlotPerEntityGlobals[entity.Type];
50 foreach (TGlobal g in array)
51 {
52 short slot = g.PerEntityIndex;
53 entityGlobals[slot] = (TGlobal)(g.InstancePerEntity ? ((GlobalType<TEntity, TGlobal>)g.NewInstance(entity)) : ((GlobalType<TEntity, TGlobal>)g));
54 }
55 setModEntityDefaults(entity);
56 EntityGlobalsEnumerator<TGlobal> entityGlobalsEnumerator = new EntityGlobalsEnumerator<TGlobal>(HookSetDefaultsEarly[entity.Type], entity);
57 EntityGlobalsEnumerator<TGlobal> enumerator = entityGlobalsEnumerator.GetEnumerator();
58 while (enumerator.MoveNext())
59 {
60 enumerator.Current.SetDefaults(entity);
61 }
62 entityGlobalsEnumerator = new EntityGlobalsEnumerator<TGlobal>(HookSetDefaultsLate[entity.Type], entity);
63 enumerator = entityGlobalsEnumerator.GetEnumerator();
64 while (enumerator.MoveNext())
65 {
66 enumerator.Current.SetDefaults(entity);
67 }
68 if (entity.Type == initialType)
69 {
70 return;
71 }
72 throw new Exception($"A mod attempted to {typeof(TEntity).Name}.type from {initialType} to {entity.Type} during SetDefaults. {entity}");
73 }

134 public static void BuildTypeLookups(Action<int> setDefaults)
135 {
136 try
137 {
138 TGlobal[] hookSetDefaults = ((IEnumerable<TGlobal>)Globals).WhereMethodIsOverridden((Expression<Func<TGlobal, Action<TEntity>>>)((TGlobal g) => g.SetDefaults)).ToArray();
139 int typeCount = GlobalList<TGlobal>.EntityTypeCount;
140 Array.Fill(HookSetDefaultsEarly = new TGlobal[typeCount][], Array.Empty<TGlobal>());
141 Array.Fill(HookSetDefaultsLate = new TGlobal[typeCount][], Array.Empty<TGlobal>());
142 TGlobal[][] globalsForType = new TGlobal[typeCount][];
143 Array.Fill(globalsForType, Array.Empty<TGlobal>());
144 GlobalTypeLookups<TGlobal>.AppliesToTypeSet[] appliesToTypeCache = new GlobalTypeLookups<TGlobal>.AppliesToTypeSet[Globals.Length];
145 InstantiationTime[] instTimes = new InstantiationTime[Globals.Length];
146 for (int setDefaultsType = 0; setDefaultsType < typeCount; setDefaultsType++)
147 {
148 int finalType = 0;
149 UpdateGlobalTypeData = delegate(int type, ReadOnlySpan<InstantiationTime> data)
150 {
151 if (type != 0)
152 {
153 finalType = type;
154 data.CopyTo(instTimes);
155 }
156 };
157 setDefaults(setDefaultsType);
158 if (finalType == 0)
159 {
160 continue;
161 }
162 globalsForType[finalType] = GlobalTypeLookups<TGlobal>.CachedFilter(Globals, (TGlobal g) => instTimes[g.StaticIndex] > InstantiationTime.NotApplied);
163 HookSetDefaultsEarly[finalType] = GlobalTypeLookups<TGlobal>.CachedFilter(hookSetDefaults, (TGlobal g) => instTimes[g.StaticIndex] == InstantiationTime.Pass1);
164 HookSetDefaultsLate[finalType] = GlobalTypeLookups<TGlobal>.CachedFilter(hookSetDefaults, (TGlobal g) => instTimes[g.StaticIndex] == InstantiationTime.Pass2);
165 TGlobal[] globals = Globals;
166 foreach (TGlobal g2 in globals)
167 {
168 if (g2.ConditionallyAppliesToEntities && instTimes[g2.StaticIndex] > InstantiationTime.NotApplied)
169 {
170 appliesToTypeCache[g2.StaticIndex].Add(finalType);
171 }
172 }
173 }
174 GlobalTypeLookups<TGlobal>.Init(globalsForType, appliesToTypeCache);
175 SlotPerEntityGlobals = GlobalTypeLookups<TGlobal>.BuildPerTypeGlobalLists(Globals.Where((TGlobal g) => g.SlotPerEntity).ToArray());
176 }
177 finally
178 {
179 UpdateGlobalTypeData = null;
180 }
181 }
