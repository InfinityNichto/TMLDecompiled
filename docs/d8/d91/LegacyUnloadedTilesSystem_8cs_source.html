164 internal void ConvertTiles()
165 {
166 List<TileEntry> legacyEntries = TileIO.Tiles.entries.ToList();
167 PosData<ushort>.OrderedSparseLookupBuilder builder = new PosData<ushort>.OrderedSparseLookupBuilder();
168 PosData<ushort>.OrderedSparseLookupReader unloadedReader = new PosData<ushort>.OrderedSparseLookupReader(TileIO.Tiles.unloadedEntryLookup);
169 for (int x = 0; x < Main.maxTilesX; x++)
170 {
171 for (int y = 0; y < Main.maxTilesY; y++)
172 {
173 Tile tile = Main.tile[x, y];
174 if (!tile.active() || tile.type < TileID.Count)
175 {
176 continue;
177 }
178 ushort type = tile.type;
179 if (TileIO.Tiles.entries[type] == null)
180 {
181 type = unloadedReader.Get(x, y);
182 if (legacyEntries[type].modName.Equals("ModLoader"))
183 {
184 ConvertTile(tile, legacyEntries, out type);
185 }
186 builder.Add(x, y, type);
187 }
188 }
189 }
190 TileIO.Tiles.entries = legacyEntries.ToArray();
191 TileIO.Tiles.unloadedEntryLookup = builder.Build();
192 }

194 internal void ConvertTile(Tile tile, List<TileEntry> entries, out ushort type)
195 {
196 int frameID = new TileFrame(tile.frameX, tile.frameY).FrameID;
197 if (!converted.TryGetValue(frameID, out type))
198 {
199 TileInfo info = infos[frameID];
200 TileEntry entry = new TileEntry(TileLoader.GetTile(tile.type))
201 {
202 name = info.name,
203 modName = info.modName,
204 frameImportant = (info.frameX > -1),
205 type = (type = (ushort)entries.Count)
206 };
207 entries.Add(entry);
208 converted.Add(frameID, type);
209 }
210 }
