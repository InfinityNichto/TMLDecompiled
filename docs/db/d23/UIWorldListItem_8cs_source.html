348 private void PlayGame(UIMouseEvent evt, UIElement listeningElement)
349 {

386 private bool TryMovingToRejectionMenuIfNeeded(int worldGameMode)
387 {
388 if (!Main.RegisteredGameModes.TryGetValue(worldGameMode, out var value))
389 {
390 SoundEngine.PlaySound(10);
391 Main.statusText = Language.GetTextValue("UI.WorldCannotBeLoadedBecauseItHasAnInvalidGameMode");
392 Main.menuMode = 1000000;
393 return true;
394 }
395 if (_canBePlayed)
396 {
397 return false;
398 }
399 bool flag = Main.ActivePlayerFileData.Player.difficulty == 3;
400 bool isJourneyMode = value.IsJourneyMode;
401 if (flag && !isJourneyMode)
402 {
403 SoundEngine.PlaySound(10);
404 Main.statusText = Language.GetTextValue("UI.PlayerIsCreativeAndWorldIsNotCreative");
405 Main.menuMode = 1000000;
406 return true;
407 }
408 if (!flag && isJourneyMode)
409 {
410 SoundEngine.PlaySound(10);
411 Main.statusText = Language.GetTextValue("UI.PlayerIsNotCreativeAndWorldIsCreative");
412 Main.menuMode = 1000000;
413 return true;
414 }
415 if (!SystemLoader.CanWorldBePlayed(Main.ActivePlayerFileData, _data, out var rejector))
416 {
417 SoundEngine.PlaySound(10);
418 Main.statusText = rejector.WorldCanBePlayedRejectionMessage(Main.ActivePlayerFileData, _data);
419 Main.menuMode = 1000000;
420 return true;
421 }
422 return false;
423 }
