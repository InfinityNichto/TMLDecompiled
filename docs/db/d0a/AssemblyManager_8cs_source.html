238 private static void VerifyMod(string modName, Assembly assembly, out Type modType)
239 {
240 string asmName = new AssemblyName(assembly.FullName).Name;
241 if (asmName != modName)
242 {
243 throw new Exception(Language.GetTextValue("tModLoader.BuildErrorModNameDoesntMatchAssemblyName", modName, asmName));
244 }
245 if (!GetLoadableTypes(assembly).Any((Type t) => t.Namespace?.StartsWith(modName) ?? false))
246 {
247 throw new Exception(Language.GetTextValue("tModLoader.BuildErrorNamespaceFolderDontMatch"));
248 }
249 Type[] modTypes = (from t in GetLoadableTypes(assembly)
250 where t.IsSubclassOf(typeof(Mod))
251 select t).ToArray();
252 if (modTypes.Length > 1)
253 {
254 throw new Exception(modName + " has multiple classes extending Mod. Only one Mod per mod is supported at the moment");
255 }
256 modType = modTypes.SingleOrDefault() ?? typeof(Mod);

259 internal static List<Mod> InstantiateMods(List<LocalMod> modsToLoad, CancellationToken token)

267 foreach (ModLoadContext mod3 in modList)

273 mod3.AddDependency(dep);

275 }
276 }

279 ModCompile.activelyModding = true;

283 Interface.loadMods.SetLoadStage("tModLoader.MSSandboxing", modsToLoad.Count);
284 int i = 0;
285 foreach (ModLoadContext mod2 in modList)

287 token.ThrowIfCancellationRequested();
288 Interface.loadMods.SetCurrentMod(i++, mod2.Name, mod2.properties?.displayName ?? "", mod2.modFile.Version);

295 Interface.loadMods.SetLoadStage("tModLoader.MSInstantiating");

417 public static void JITAssemblies(IEnumerable<Assembly> assemblies, PreJITFilter filter)

422 MethodBase[] methodsToJIT = GetLoadableTypes(assembly).Where(filter.ShouldJIT).SelectMany((Type type) => (from m in ((IEnumerable<MethodBase>)(from m in type.GetMethods(BindingFlags.Instance | BindingFlags.Static | BindingFlags.Public | BindingFlags.NonPublic)
423 where !m.IsSpecialName
424 select m)).Concat((IEnumerable<MethodBase>)type.GetConstructors(BindingFlags.Instance | BindingFlags.Static | BindingFlags.Public | BindingFlags.NonPublic)).Concat(type.GetProperties(BindingFlags.Instance | BindingFlags.Static | BindingFlags.Public | BindingFlags.NonPublic).Where(filter.ShouldJIT).SelectMany((PropertyInfo p) => p.GetAccessors()))
425 where !m.IsAbstract && !m.ContainsGenericParameters && m.DeclaringType == type
426 select m).Where(filter.ShouldJIT)).ToArray();

429 methodsToJIT.AsParallel().AsUnordered().ForAll(delegate(MethodBase method)

431 try

433 ForceJITOnMethod(method);
434 }
435 catch (Exception item)

437 exceptions.Enqueue((item, method));

461 private static void ForceJITOnMethod(MethodBase method)
462 {

475 if (flag.HasValue && !flag.GetValueOrDefault() && methodInfo.GetBaseDefinition() == methodInfo)
