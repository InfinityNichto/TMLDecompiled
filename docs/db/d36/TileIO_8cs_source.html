436 internal static void ReadContainers(BinaryReader reader)
437 {
438 byte[] flags = new byte[1];
439 reader.Read(flags, 0, reader.ReadByte());
440 if ((flags[0] & 1) == 1)
441 {
442 ContainerTables tables = ContainerTables.Create();
443 int count = reader.ReadUInt16();
444 for (int k = 0; k < count; k++)
445 {
446 tables.headSlots[reader.ReadUInt16()] = (ModContent.TryFind<ModItem>(reader.ReadString(), reader.ReadString(), out var item3) ? item3.Item.headSlot : 0);
447 }
448 count = reader.ReadUInt16();
449 for (int j = 0; j < count; j++)
450 {
451 tables.bodySlots[reader.ReadUInt16()] = (ModContent.TryFind<ModItem>(reader.ReadString(), reader.ReadString(), out var item2) ? item2.Item.bodySlot : 0);
452 }
453 count = reader.ReadUInt16();
454 for (int i = 0; i < count; i++)
455 {
456 tables.legSlots[reader.ReadUInt16()] = (ModContent.TryFind<ModItem>(reader.ReadString(), reader.ReadString(), out var item) ? item.Item.legSlot : 0);

462 internal static void WriteContainerData(BinaryWriter writer)
463 {
464 for (int i = 0; i < Main.maxTilesX; i++)
465 {
466 for (int j = 0; j < Main.maxTilesY; j++)
467 {
468 Tile tile = Main.tile[i, j];
469 if (tile.active() && (tile.type == 128 || tile.type == 269))
470 {
471 int slot = tile.frameX / 100;
472 int frameX = tile.frameX % 100;
473 int position = tile.frameY / 18;
474 if (HasModArmor(slot, position) && frameX % 36 == 0)
475 {
476 writer.Write(i);
477 writer.Write(j);
478 writer.Write((byte)position);

481 }
482 }

487 internal static void ReadContainerData(BinaryReader reader, ContainerTables tables)
488 {
489 for (int i = reader.ReadInt32(); i > 0; i = reader.ReadInt32())
490 {
491 int j = reader.ReadInt32();
492 int position = reader.ReadByte();
493 int slot = reader.ReadUInt16();
494 Tile left = Main.tile[i, j];
495 Tile right = Main.tile[i + 1, j];
496 if (left.active() && right.active() && (left.type == 128 || left.type == 269) && left.type == right.type && (left.frameX == 0 || left.frameX == 36) && right.frameX == left.frameX + 18 && left.frameY / 18 == position && left.frameY == right.frameY)
497 {
498 switch (position)
499 {
500 case 0:
501 slot = tables.headSlots[slot];
502 break;
503 case 1:
504 slot = tables.bodySlots[slot];
505 break;
506 case 2:
507 slot = tables.legSlots[slot];
508 break;
509 }
510 left.frameX += (short)(100 * slot);
511 }
512 }
