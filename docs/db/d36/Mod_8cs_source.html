130 internal void AutoloadConfig()
131 {
132 if (Code == null)
133 {
134 return;
135 }
136 foreach (Type type2 in from type in AssemblyManager.GetLoadableTypes(Code)
137 orderby type.FullName
138 select type)
139 {
140 if (!type2.IsAbstract && type2.IsSubclassOf(typeof(ModConfig)))
141 {
142 ModConfig mc = (ModConfig)Activator.CreateInstance(type2, nonPublic: true);
143 if (mc.Mode == ConfigScope.ServerSide && (Side == ModSide.Client || Side == ModSide.NoSync))
144 {
145 throw new Exception("The ModConfig " + mc.Name + " can't be loaded because the config is ServerSide but this Mods ModSide isn't Both or Server");
146 }
147 if (mc.Mode == ConfigScope.ClientSide && Side == ModSide.Server)
148 {
149 throw new Exception("The ModConfig " + mc.Name + " can't be loaded because the config is ClientSide but this Mods ModSide is Server");
150 }
151 mc.Mod = this;
152 string name = type2.Name;
153 if (mc.Autoload(ref name))
154 {
155 AddConfig(name, mc);
156 }
157 }

504 internal void Autoload()
505 {
506 if (Code == null)
507 {
508 return;
509 }
510 LocalizationLoader.Autoload(this);
511 Interface.loadMods.SubProgressText = Language.GetTextValue("tModLoader.MSFinishingResourceLoading");
512 while (AsyncLoadQueue.Count > 0)
513 {
514 AsyncLoadQueue.Dequeue().Wait();
515 }
516 ModSourceBestiaryInfoElement = new ModSourceBestiaryInfoElement(this, DisplayName);
517 if (ContentAutoloadingEnabled)
518 {
519 LoaderUtils.ForEachAndAggregateExceptions((from t in AssemblyManager.GetLoadableTypes(Code)
520 where !t.IsAbstract && !t.ContainsGenericParameters
521 where t.IsAssignableTo(typeof(ILoadable))
522 where t.GetConstructor(BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic, Type.EmptyTypes) != null
523 where AutoloadAttribute.GetValue(t).NeedsAutoloading
524 select t).OrderBy<Type, string>((Type type) => type.FullName, StringComparer.InvariantCulture), delegate(Type t)
525 {
526 AddContent((ILoadable)Activator.CreateInstance(t, nonPublic: true));
527 });
528 }
529 if (!Main.dedServ)
530 {
531 if (GoreAutoloadingEnabled)
532 {
533 GoreLoader.AutoloadGores(this);
534 }
535 if (MusicAutoloadingEnabled)
536 {
537 MusicLoader.AutoloadMusic(this);
538 }
539 if (BackgroundAutoloadingEnabled)
540 {
541 BackgroundTextureLoader.AutoloadBackgrounds(this);
542 }
543 }

574 internal void OnceFailedLoadingAnAsset(string assetPath, Exception e)
575 {
576 if (initialTransferComplete)
577 {
578 Logging.Terraria.Error((object)("Failed to load asset: \"" + assetPath + "\""), e);
579 FancyErrorPrinter.ShowFailedToLoadAssetError(e, assetPath);
580 }
581 else if (e is AssetLoadException)
582 {
583 List<string> list = RootContentSource.EnumerateAssets().ToList();
584 List<string> cleanKeys = new List<string>();
585 foreach (string key in (IEnumerable<string>)list)
586 {
587 string keyWithoutExtension = key.Substring(0, key.LastIndexOf("."));
588 string extension = RootContentSource.GetExtension(keyWithoutExtension);
589 if (extension != null)
590 {
591 cleanKeys.Add(key.Substring(0, key.LastIndexOf(extension)));
592 }
593 }
594 List<string> reasons = new List<string>();
595 RootContentSource.Rejections.TryGetRejections(reasons);
596 MissingResourceException MissingResourceException = new MissingResourceException(reasons, assetPath.Replace("\\", "/"), cleanKeys);
597 AssetExceptions.Add(MissingResourceException);
598 }
599 else
600 {
601 AssetExceptions.Add(e);
602 }
603 }
