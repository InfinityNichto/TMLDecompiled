99 {
100 TypeCloningInfo info = new TypeCloningInfo

103 overridesClone = (LoaderUtils.GetDerivedDefinition(type, cloneMethod).DeclaringType == type),
104 fieldsWhichMightNeedDeepCloning = (from f in type.GetFields(BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic)
