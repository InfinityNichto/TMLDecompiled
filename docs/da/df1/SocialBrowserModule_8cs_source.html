66 void GetDependenciesRecursive(HashSet<ModDownloadItem> set)
67 {
68 HashSet<ModPubId_t> fullList = set.Select((ModDownloadItem x) => x.PublishId).ToHashSet();
69 HashSet<ModPubId_t> iterationList = new HashSet<ModPubId_t>();
70 HashSet<ModDownloadItem> iterationSet = set;
71 while (true)
72 {
73 foreach (ModDownloadItem item in iterationSet)
74 {
75 iterationList.UnionWith(item.ModReferenceByModId);
76 }
77 iterationList.ExceptWith(fullList);
78 if (iterationList.Count <= 0)
79 {
80 break;
81 }
82 iterationSet = DirectQueryItems(new QueryParameters
83 {
84 searchModIds = iterationList.ToArray()
85 }, out var notFoundMods).ToHashSet();
86 if (notFoundMods.Any())
87 {
88 notFoundMods = notFoundMods;
89 }
90 fullList.UnionWith(iterationList);
91 set.UnionWith(iterationSet);
92 }
93 }
