295 public static object Deserialize(Type type, object? tag)
296 {
297 ArgumentNullException.ThrowIfNull(type, "type");
298 if (type.IsInstanceOfType(tag))
299 {
300 return tag;
301 }
302 if (TagSerializer.TryGetSerializer(type, out TagSerializer serializer))
303 {
304 if (tag == null)
305 {
306 tag = Deserialize(serializer.TagType, null);
307 }
308 return serializer.Deserialize(tag);
309 }
310 if (tag == null && !type.IsArray)
311 {
312 if (type.GetGenericArguments().Length == 0)
313 {
314 return GetHandler(GetPayloadId(type)).Default();
315 }
316 if (type.GetGenericTypeDefinition() == typeof(Nullable<>))
317 {
318 return Activator.CreateInstance(type);
319 }
320 }
321 if (tag == null || tag is IList || type.IsArray)
322 {
323 if (type.IsArray)
324 {
325 Type elemType = type.GetElementType();
326 if (tag == null)
327 {
328 return Array.CreateInstance(elemType, 0);
329 }
330 IList serializedList = (IList)tag;
331 if (TagSerializer.TryGetSerializer(elemType, out serializer))
332 {
333 IList array = Array.CreateInstance(elemType, serializedList.Count);
334 for (int i = 0; i < serializedList.Count; i++)
335 {
336 array[i] = serializer.Deserialize(serializedList[i]);
337 }
338 return array;
339 }
340 IList deserializedArray = Array.CreateInstance(elemType, serializedList.Count);
341 for (int j = 0; j < serializedList.Count; j++)
342 {
343 deserializedArray[j] = Deserialize(elemType, serializedList[j]);
344 }
345 return deserializedArray;
346 }
347 if (type.GetGenericArguments().Length == 1)
348 {
349 Type elemType2 = type.GetGenericArguments()[0];

351 if (type.IsAssignableFrom(newListType))
