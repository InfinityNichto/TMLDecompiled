380 private static void FirstChanceExceptionHandler(object sender, FirstChanceExceptionEventArgs args)
381 {
382 if (quietExceptionCount > 0)
383 {
384 return;
385 }
386 using (new QuietExceptionHandle())
387 {
388 bool oom = args.Exception is OutOfMemoryException;
389 if (oom)
390 {
391 TryFreeingMemory();
392 }
393 try
394 {
395 if (!oom && (args.Exception == previousException || args.Exception is ThreadAbortException || ignoreTypes.Contains(args.Exception.GetType().FullName) || ignoreSources.Contains(args.Exception.Source) || ignoreMessages.Any((string str) => args.Exception.Message?.Contains(str) ?? false) || ignoreThrowingMethods.Any((string str) => args.Exception.StackTrace?.Contains(str) ?? false)))
396 {
397 return;
398 }
399 StackTrace stackTrace = new StackTrace(1, fNeedFileInfo: true);
400 string traceString = stackTrace.ToString();
401 if (!oom && ignoreContents.Any((string s) => MatchContents(traceString, s)))
402 {
403 return;
404 }
405 traceString = stackTrace.ToString();
406 string text = traceString;
407 int num = traceString.IndexOf('\n');
408 traceString = text.Substring(num, text.Length - num);
409 string exString = args.Exception.GetType()?.ToString() + ": " + args.Exception.Message + traceString;
410 lock (pastExceptions)
411 {
412 if (!pastExceptions.Add(exString))
413 {
414 return;
415 }
416 }
417 tML.Warn((object)(Language.GetTextValue("tModLoader.RuntimeErrorSilentlyCaughtException") + "\n" + exString));
418 previousException = args.Exception;

443 {
444 while (true)

447 ReadOnlySpan<char> readOnlySpan;
