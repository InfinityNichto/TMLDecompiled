351 internal static void ResizeArrays(bool unloading)
352 {
353 if (!unloading)
354 {
355 GlobalList<GlobalItem>.FinishLoading(ItemCount);
356 }
357 Array.Resize(ref TextureAssets.Item, ItemCount);
358 Array.Resize(ref TextureAssets.ItemFlame, ItemCount);
359 LoaderUtils.ResetStaticMembers(typeof(ItemID));
360 LoaderUtils.ResetStaticMembers(typeof(AmmoID));
361 LoaderUtils.ResetStaticMembers(typeof(PrefixLegacy.ItemSets));
362 Array.Resize(ref Item.cachedItemSpawnsByType, ItemCount);
363 Array.Resize(ref Item.staff, ItemCount);
364 Array.Resize(ref Item.claw, ItemCount);
365 Array.Resize(ref Lang._itemNameCache, ItemCount);
366 Array.Resize(ref Lang._itemTooltipCache, ItemCount);
367 Array.Resize(ref RecipeLoader.FirstRecipeForItem, ItemCount);
368 for (int i = ItemID.Count; i < ItemCount; i++)
369 {
370 Lang._itemNameCache[i] = LocalizedText.Empty;
371 Lang._itemTooltipCache[i] = ItemTooltip.None;
372 Item.cachedItemSpawnsByType[i] = -1;
373 }
374 lock (Main.itemAnimationsRegistered)
375 {
376 Array.Resize(ref Main.itemAnimations, ItemCount);
377 Main.InitializeItemAnimations();
378 }
379 if (unloading)
380 {
381 Array.Resize(ref Main.anglerQuestItemNetIDs, vanillaQuestFishCount);
382 return;
383 }
384 Main.anglerQuestItemNetIDs = Main.anglerQuestItemNetIDs.Concat(from modItem in items
385 where modItem.IsQuestFish()
386 select modItem.Type).ToArray();
387 }
