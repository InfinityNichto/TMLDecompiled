92 private static void Resize2DArray<T>(ref T[,] array, int newSize)
93 {
94 int dim1 = array.GetLength(0);
95 int dim2 = array.GetLength(1);
96 T[,] newArray = new T[newSize, dim2];
97 for (int i = 0; i < newSize && i < dim1; i++)

99 for (int j = 0; j < dim2; j++)
